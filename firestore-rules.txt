rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Rules for the 'enquiries' collection
    match /enquiries/{enquiryId} {
      
      // ALLOW anyone to create a new enquiry (public contact form)
      allow create: if true;
      
      // DENY reading, updating, or deleting from the frontend
      // (You will still be able to read them perfectly from the Firebase Console)
      allow read, update, delete: if false;
    }
    
    // Rules for the 'config' collection (Admin Auth)
    match /config/admin {
      // Disallow all writes from frontend - manage this manually in console
      allow write: if false;
      // Allow the admin login page to fetch the password for verification
      allow read: if true;
    }
    
    // Deny access to all other collections by default for security
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
